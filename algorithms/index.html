from collections import Iterable

def flatten(input_arr, output_arr=None):
    if output_arr is None:
        output_arr = []
    for elem in input_arr:
        if isinstance(elem, Iterable):
            flatten(elem, output_arr)
        else:
            output_arr.append(elem)
    return output_arr
