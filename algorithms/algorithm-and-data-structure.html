def binary_search(arr, target):
    """
    :type arr:list[int/float/str] AESC
    :type target: int/float/str
    :rtype int: arr[target].index()
    """
    l, r = 0, len(arr)-1
    while l <= r:
        mid = (l + r) // 2
        if target == arr[mid]:
            return mid
        if target > arr[mid]:
            l = mid + 1
        else:
            r = mid - 1
    return -1
